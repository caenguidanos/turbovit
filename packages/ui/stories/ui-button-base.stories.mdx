# Button

```ts
import { Button, ButtonProps } from "ui-button";

export const App: React.FunctionComponent = () => {
   let props: ButtonProps = {
      text: "Click me!",
      color: "primary",
      size: "medium",
   };

   const handleOnClick = (ev) => {
      console.log(ev);
   };

   return <Button {...props} onClick={handleOnClick} />;
};
```
